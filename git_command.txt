git init - инициализация пустого репозитория

git remote -v - посмотреть удаленые репозитории
git remote add origin "сслыка на удаленый рупозиторий" - привязка удаленного репозитория

git clone "Ссылка на репозиторий" - клонирует репозиторий в текущую папку и задает имя папки именем репозитория.
git clone "Ссылка на репозиторий" . -  клонирует репозиторий в текущую папку
git clone "Ссылка на репозиторий" "Имя папки" - клонирует репозиторий в текущую папку с заданным именем

git status - показывает состояние репозитория

git add "File name" - добавить файл в рабочее дерево репозитория
git add . - добавить в дерево все изменения всех файлов

git commit -m "комментарий" - подтвердить изменения

git push origin "имя ветки" - отправить изменения в удаленный репозиторий

git log - посмотреть историю изменений // q + enter - выйти

git branch "название новой ветки" - создание новой ветки
git branch -d "название ветки" - удалить ветку, нужно находиться на другой ветки для удаления
git branch - показывает список существующих веток
git branch -m "новое название ветки" - переименовывает текущую ветку в которой находимся
git checkout "название ветки" - переключиться на ветку
git checkout -b "название ветки" - создание новой ветки и переход на нее
